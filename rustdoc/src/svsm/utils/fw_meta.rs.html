7use crate::types::PAGE_SIZE;
8use uuid::Uuid;
9use zerocopy::{FromBytes, Immutable, KnownLayout};
10
11use core::mem::{size_of, size_of_val};
12
13#[derive(Debug, Clone, Copy, FromBytes, KnownLayout, Immutable)]
14#[repr(C, packed)]
15pub struct RawMetaHeader {
16    len: u16,
17    uuid: [u8; size_of::<Uuid>()],
18}
19
20impl RawMetaHeader {
21    pub const fn uuid(&self) -> Uuid {
22        Uuid::from_bytes_le(self.uuid)
23    }
24
25    pub fn data_len(&self) -> Option<usize> {
26        let full_len = self.len as usize;
27        full_len.checked_sub(size_of::<Self>())
28    }
29}
30
31#[derive(Debug, Clone, Copy, FromBytes, KnownLayout, Immutable)]
32#[repr(C, packed)]
33pub struct RawMetaBuffer {
34    pub data: [u8; PAGE_SIZE - size_of::<RawMetaHeader>() - 32],
35    pub header: RawMetaHeader,
36    _pad: [u8; 32],
37}
38
39impl RawMetaBuffer {
40    pub fn pad_size(&self) -> usize {
41        size_of_val(&self._pad)
42    }
43}
44
45// Compile-time size checks
46const _: () = assert!(size_of::<RawMetaBuffer>() == PAGE_SIZE);
47const _: () = assert!(size_of::<RawMetaHeader>() == size_of::<u16>() + size_of::<Uuid>());
48
49/// Find a table with the given UUID in the given memory slice, and return a
50/// subslice into its data
51pub fn find_table<'a>(uuid: &Uuid, mem: &'a [u8]) -> Option<&'a [u8]> {
52    let mut idx = mem.len();
53
54    while idx != 0 {
55        let hdr_start = idx.checked_sub(size_of::<RawMetaHeader>())?;
56        let hdr = RawMetaHeader::ref_from_bytes(&mem[hdr_start..idx]).unwrap();
57
58        let data_len = hdr.data_len()?;
59        idx = hdr_start.checked_sub(data_len)?;
60
61        let raw_uuid = hdr.uuid;
62        let curr_uuid = Uuid::from_bytes_le(raw_uuid);
63        if *uuid == curr_uuid {
64            return Some(&mem[idx..idx + data_len]);
65        }
66    }
67
68    None
69}
