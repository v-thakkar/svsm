44// This lock prevents races around PVALIDATE and CREATE_VCPU
45//
46// Without the lock there is a possible attack where the error path of
47// core_create_vcpu() could give the guest OS access to a SVSM page.
48//
49// The PValidate path will take the lock for read, the create_vcpu path takes
50// the lock for write.
51static PVALIDATE_LOCK: RWLock<()> = RWLock::new(());
52
53#[repr(C, packed)]
54#[derive(Copy, Clone)]
55struct PValidateRequest {
56    entries: u16,
57    next: u16,
58    resv: u32,
59}

61fn core_create_vcpu_error_restore(paddr: Option<PhysAddr>, vaddr: Option<VirtAddr>) {
62    if let Some(v) = vaddr {
63        if let Err(err) = rmp_clear_guest_vmsa(v) {
64            log::error!("Failed to restore page permissions: {:#?}", err);
65        }
66    }
67    // In case mappings have been changed
68    flush_tlb_global_sync();
69
70    if let Some(p) = paddr {
71        // SAFETY: This can only fail if another CPU unregisters our
72        // unused VMSA. This is not possible, since unregistration of
73        // an unused VMSA only happens in the error path of core_create_vcpu(),
74        // with a physical address that only this CPU managed to register.
75        PERCPU_VMSAS.unregister(p, false).unwrap();
76    }
77}

79// VMSA validity checks according to SVSM spec
80fn check_vmsa(new: &VMSA, sev_features: u64, svme_mask: u64) -> bool {
81    new.vmpl == RMPFlags::GUEST_VMPL.bits() as u8
82        && new.efer & svme_mask == svme_mask
83        && new.sev_features == sev_features
84}

86/// per-cpu request mapping area size (1GB)
87fn core_create_vcpu(params: &RequestParams) -> Result<(), SvsmReqError> {
88    let paddr = PhysAddr::from(params.rcx);
89    let pcaa = PhysAddr::from(params.rdx);
90    let apic_id: u32 = (params.r8 & 0xffff_ffff) as u32;
91
92    // Check VMSA address
93    if !valid_phys_address(paddr) || !paddr.is_page_aligned() {
94        return Err(SvsmReqError::invalid_address());
95    }
96
97    // Check CAA address
98    if !valid_phys_address(pcaa) || !pcaa.is_page_aligned() {
99        return Err(SvsmReqError::invalid_address());
100    }
101
102    // Check whether VMSA page and CAA region overlap
103    //
104    // Since both areas are 4kb aligned and 4kb in size, and correct alignment
105    // was already checked, it is enough here to check whether VMSA and CAA
106    // page have the same starting address.
107    if paddr == pcaa {
108        return Err(SvsmReqError::invalid_address());
109    }
110
111    let target_cpu = PERCPU_AREAS
112        .get_by_apic_id(apic_id)
113        .ok_or_else(SvsmReqError::invalid_parameter)?;
114
115    // Got valid gPAs and APIC ID, register VMSA immediately to avoid races
116    PERCPU_VMSAS.register(paddr, target_cpu.cpu_index(), true)?;
117
118    // Time to map the VMSA. No need to clean up the registered VMSA on the
119    // error path since this is a fatal error anyway.
120    let mapping_guard = PerCPUPageMappingGuard::create_4k(paddr)?;
121    let vaddr = mapping_guard.virt_addr();
122
123    // Prevent any parallel PVALIDATE requests from being processed
124    let lock = PVALIDATE_LOCK.lock_write();
125
126    // Make sure the guest can't make modifications to the VMSA page
127    rmp_revoke_guest_access(vaddr, PageSize::Regular).inspect_err(|_| {
128        core_create_vcpu_error_restore(Some(paddr), None);
129    })?;
130
131    // TLB flush needed to propagate new permissions
132    flush_tlb_global_sync();
133
134    let new_vmsa = vmsa_ref_from_vaddr(vaddr);
135    let svme_mask: u64 = 1u64 << 12;
136
137    // VMSA validity checks according to SVSM spec
138    if !check_vmsa(new_vmsa, params.sev_features, svme_mask) {
139        core_create_vcpu_error_restore(Some(paddr), Some(vaddr));
140        return Err(SvsmReqError::invalid_parameter());
141    }
142
143    // Set the VMSA bit
144    rmp_set_guest_vmsa(vaddr).inspect_err(|_| {
145        core_create_vcpu_error_restore(Some(paddr), Some(vaddr));
146    })?;
147
148    drop(lock);
149
150    assert!(PERCPU_VMSAS.set_used(paddr) == Some(target_cpu.cpu_index()));
151    target_cpu.update_guest_vmsa_caa(paddr, pcaa);
152
153    Ok(())
154}

156fn core_delete_vcpu(params: &RequestParams) -> Result<(), SvsmReqError> {
157    let paddr = PhysAddr::from(params.rcx);
158
159    PERCPU_VMSAS
160        .unregister(paddr, true)
161        .map_err(|_| SvsmReqError::invalid_parameter())?;
162
163    // Map the VMSA
164    let mapping_guard = PerCPUPageMappingGuard::create_4k(paddr)?;
165    let vaddr = mapping_guard.virt_addr();
166
167    // Clear EFER.SVME on deleted VMSA. If the VMSA is executing
168    // disable() will loop until that is not the case
169    let del_vmsa = vmsa_mut_ref_from_vaddr(vaddr);
170    del_vmsa.disable();
171
172    // Do not return early here, as we need to do a TLB flush
173    let res = rmp_clear_guest_vmsa(vaddr).map_err(|_| SvsmReqError::invalid_address());
174
175    // Unmap the page
176    drop(mapping_guard);
177
178    // Tell everyone the news and flush temporary mapping
179    flush_tlb_global_sync();
180
181    res
182}

194fn protocol_supported(version: u32, version_min: u32, version_max: u32) -> u64 {
195    if version >= version_min && version <= version_max {
196        let ret_low: u64 = version_min.into();
197        let ret_high: u64 = version_max.into();
198
199        ret_low | (ret_high << 32)
200    } else {
201        0
202    }
203}

205fn core_query_protocol(params: &mut RequestParams) -> Result<(), SvsmReqError> {
206    let rcx: u64 = params.rcx;
207    let protocol: u32 = (rcx >> 32).try_into().unwrap();
208    let version: u32 = (rcx & 0xffff_ffffu64).try_into().unwrap();
209
210    let ret_val = match protocol {
211        SVSM_CORE_PROTOCOL => protocol_supported(
212            version,
213            CORE_PROTOCOL_VERSION_MIN,
214            CORE_PROTOCOL_VERSION_MAX,
215        ),
216        SVSM_APIC_PROTOCOL => {
217            // The APIC protocol is only supported if the calling CPU supports
218            // alternate injection.
219            if this_cpu().use_apic_emulation() {
220                protocol_supported(
221                    version,
222                    APIC_PROTOCOL_VERSION_MIN,
223                    APIC_PROTOCOL_VERSION_MAX,
224                )
225            } else {
226                0
227            }
228        }
229        SVSM_ATTEST_PROTOCOL => protocol_supported(
230            version,
231            ATTEST_PROTOCOL_VERSION_MIN,
232            ATTEST_PROTOCOL_VERSION_MAX,
233        ),
234        _ => 0,
235    };
236
237    params.rcx = ret_val;
238
239    Ok(())
240}

254fn core_pvalidate_one(entry: u64, flush: &mut bool) -> Result<(), SvsmReqError> {
255    let (page_size_bytes, valign, huge) = match entry & 3 {
256        0 => (PAGE_SIZE, VIRT_ALIGN_4K, PageSize::Regular),
257        1 => (PAGE_SIZE_2M, VIRT_ALIGN_2M, PageSize::Huge),
258        _ => return Err(SvsmReqError::invalid_parameter()),
259    };
260
261    let valid = match (entry & 4) == 4 {
262        true => PvalidateOp::Valid,
263        false => PvalidateOp::Invalid,
264    };
265    let ign_cf = (entry & 8) == 8;
266
267    let paddr = PhysAddr::from(entry).page_align();
268
269    if !paddr.is_aligned(page_size_bytes) {
270        return Err(SvsmReqError::invalid_parameter());
271    }
272
273    if !valid_phys_address(paddr) {
274        log::debug!("Invalid phys address: {:#x}", paddr);
275        return Err(SvsmReqError::invalid_address());
276    }
277
278    let guard = PerCPUPageMappingGuard::create(paddr, paddr + page_size_bytes, valign)?;
279    let vaddr = guard.virt_addr();
280
281    // Take lock to prevent races with CREATE_VCPU calls
282    let lock = PVALIDATE_LOCK.lock_read();
283
284    if valid == PvalidateOp::Invalid {
285        *flush |= true;
286        rmp_revoke_guest_access(vaddr, huge)?;
287    }
288
289    pvalidate(vaddr, huge, valid).or_else(|err| match err {
290        SvsmError::SevSnp(SevSnpError::FAIL_UNCHANGED(_)) if ign_cf => Ok(()),
291        _ => Err(err),
292    })?;
293
294    drop(lock);
295
296    if valid == PvalidateOp::Valid {
297        // Zero out a page when it is validated and before giving other VMPLs
298        // access to it. This is necessary to prevent a possible HV attack:
299        //
300        // Attack scenario:
301        //   1) SVSM stores secrets in VMPL0 memory at GPA A
302        //   2) HV invalidates GPA A and maps the SPA to GPA B, which is in the
303        //      OS range of GPAs
304        //   3) Guest OS asks SVSM to validate GPA B
305        //   4) SVSM validates page and gives OS access
306        //   5) OS can now read SVSM secrets from GPA B
307        //
308        // The SVSM will not notice the attack until it tries to access GPA A
309        // again. Prevent it by clearing every page before giving access to
310        // other VMPLs.
311        //
312        // Be careful to not clear GPAs which the HV might have mapped
313        // read-only, as the write operation might cause infinite #NPF loops.
314        //
315        // Special thanks to Tom Lendacky for reporting the issue and tracking
316        // down the #NPF loops.
317        //
318        if writable_phys_addr(paddr) {
319            // FIXME: This check leaves a window open for the attack described
320            // above. Remove the check once OVMF and Linux have been fixed and
321            // no longer try to pvalidate MMIO memory.
322
323            // SAFETY: paddr is validated at the beginning of the function, and
324            // we trust PerCPUPageMappingGuard::create() to return a valid
325            // vaddr pointing to a mapped region of at least page_size_bytes
326            // size.
327            unsafe {
328                zero_mem_region(vaddr, vaddr + page_size_bytes);
329            }
330        } else {
331            log::warn!("Not clearing possible read-only page at PA {:#x}", paddr);
332        }
333        rmp_grant_guest_access(vaddr, huge)?;
334    }
335
336    Ok(())
337}

339fn core_pvalidate(params: &RequestParams) -> Result<(), SvsmReqError> {
340    let gpa = PhysAddr::from(params.rcx);
341
342    if !gpa.is_aligned(8) || !valid_phys_address(gpa) {
343        return Err(SvsmReqError::invalid_parameter());
344    }
345
346    let paddr = gpa.page_align();
347    let offset = gpa.page_offset();
348
349    let guard = PerCPUPageMappingGuard::create_4k(paddr)?;
350    let start = guard.virt_addr();
351
352    let guest_page = GuestPtr::<PValidateRequest>::new(start + offset);
353    // SAFETY: start is a new mapped page address, thus valid.
354    // offset can't exceed a page size, so guest_page belongs to mapped memory.
355    let mut request = unsafe { guest_page.read()? };
356
357    let entries = request.entries;
358    let next = request.next;
359
360    // Each entry is 8 bytes in size, 8 bytes for the request header
361    let max_entries: u16 = ((PAGE_SIZE - offset - 8) / 8).try_into().unwrap();
362
363    if entries == 0 || entries > max_entries || entries <= next {
364        return Err(SvsmReqError::invalid_parameter());
365    }
366
367    let mut loop_result = Ok(());
368    let mut flush = false;
369
370    let guest_entries = guest_page.offset(1).cast::<u64>();
371    for i in next..entries {
372        let index = i as isize;
373        // SAFETY: guest_entries comes from guest_page which is a new mapped
374        // page. index is between [next, entries) and both values have been
375        // validated.
376        let entry = match unsafe { guest_entries.offset(index).read() } {
377            Ok(v) => v,
378            Err(e) => {
379                loop_result = Err(e.into());
380                break;
381            }
382        };
383
384        loop_result = core_pvalidate_one(entry, &mut flush);
385        match loop_result {
386            Ok(()) => request.next += 1,
387            Err(SvsmReqError::RequestError(..)) => break,
388            Err(SvsmReqError::FatalError(..)) => return loop_result,
389        }
390    }
391
392    // SAFETY: guest_page is obtained from a guest-provided physical address
393    // (untrusted), so it needs to be valid (ie. belongs to the guest and only
394    // the guest). The physical address is validated by valid_phys_address()
395    // called at the beginning of SVSM_CORE_PVALIDATE handler (this one).
396    if let Err(e) = unsafe { guest_page.write_ref(&request) } {
397        loop_result = Err(e.into());
398    }
399
400    if flush {
401        flush_tlb_global_sync();
402    }
403
404    loop_result
405}

407fn core_remap_ca(params: &RequestParams) -> Result<(), SvsmReqError> {
408    let gpa = PhysAddr::from(params.rcx);
409
410    if !gpa.is_aligned(8) || !valid_phys_address(gpa) || gpa.crosses_page(8) {
411        return Err(SvsmReqError::invalid_parameter());
412    }
413
414    let offset = gpa.page_offset();
415    let paddr = gpa.page_align();
416
417    // Temporarily map new CAA to clear it
418    let mapping_guard = PerCPUPageMappingGuard::create_4k(paddr)?;
419    let vaddr = mapping_guard.virt_addr() + offset;
420
421    let pending = GuestPtr::<SvsmCaa>::new(vaddr);
422    // SAFETY: pending points to a new allocated page
423    unsafe { pending.write(SvsmCaa::zeroed())? };
424
425    // Clear any pending interrupt state before remapping the calling area to
426    // ensure that any pending lazy EOI has been processed.
427    this_cpu().clear_pending_interrupts();
428
429    this_cpu_shared().update_guest_caa(gpa);
430
431    Ok(())
432}

434pub fn core_protocol_request(request: u32, params: &mut RequestParams) -> Result<(), SvsmReqError> {
435    match request {
436        SVSM_REQ_CORE_REMAP_CA => core_remap_ca(params),
437        SVSM_REQ_CORE_PVALIDATE => core_pvalidate(params),
438        SVSM_REQ_CORE_CREATE_VCPU => core_create_vcpu(params),
439        SVSM_REQ_CORE_DELETE_VCPU => core_delete_vcpu(params),
440        SVSM_REQ_CORE_DEPOSIT_MEM => core_deposit_mem(params),
441        SVSM_REQ_CORE_WITHDRAW_MEM => core_withdraw_mem(params),
442        SVSM_REQ_CORE_QUERY_PROTOCOL => core_query_protocol(params),
443        SVSM_REQ_CORE_CONFIGURE_VTOM => core_configure_vtom(params),
444        _ => Err(SvsmReqError::unsupported_call()),
445    }
