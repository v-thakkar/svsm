13use crate::utils::{page_align_up, MemoryRegion};
14

16use super::Mapping;

18/// Mapping to be used as a kernel stack. This maps a stack including guard
19/// pages at the top and bottom.
20#[derive(Default, Debug)]
21pub struct VMKernelStack {
22    /// Allocation for stack pages
23    alloc: RawAllocMapping,
24    /// Number of guard pages to reserve address space for
25    guard_pages: usize,
26}
27
28impl VMKernelStack {
29    /// Returns the virtual address for the top of this kernel stack
30    ///
31    /// # Arguments
32    ///
33    /// * `base` - Virtual base address this stack is mapped at (including
34    ///   guard pages).
35    ///
36    /// # Returns
37    ///
38    /// Virtual address to program into the hardware stack register
39    pub fn top_of_stack(&self, base: VirtAddr) -> VirtAddr {
40        let guard_size = self.guard_pages * PAGE_SIZE;
41        let tos = (base + guard_size + self.alloc.mapping_size()).align_down(16);
42        debug_assert!(tos > base + guard_size);
43        tos
44    }
45
46    /// Returns the stack bounds of this kernel stack
47    ///
48    /// # Arguments
49    ///
50    /// * `base` - Virtual base address this stack is mapped at (including
51    ///   guard pages).
52    ///
53    /// # Returns
54    ///
55    /// A [`MemoryRegion`] object containing the bottom and top addresses for
56    /// the stack
57    pub fn bounds(&self, base: VirtAddr) -> MemoryRegion<VirtAddr> {
58        let mapping_size = self.alloc.mapping_size();
59        let guard_size = self.guard_pages * PAGE_SIZE;
60        MemoryRegion::new(base + guard_size, mapping_size)
61    }
62
63    /// Create a new [`VMKernelStack`] with a given size. This function will
64    /// already allocate the backing pages for the stack.
65    ///
66    /// # Arguments
67    ///
68    /// * `size` - Size of the kernel stack, without guard pages
69    ///
70    /// # Returns
71    ///
72    /// Initialized stack on success, Err(SvsmError::Mem) on error
73    pub fn new_size(size: usize) -> Result<Self, SvsmError> {
74        // Make sure size is page-aligned
75        let size = page_align_up(size);
76        // At least two guard-pages needed
77        let total_size = (size + 2 * PAGE_SIZE).next_power_of_two();
78        let guard_pages = ((total_size - size) >> PAGE_SHIFT) / 2;
79        let mut stack = VMKernelStack {
80            alloc: RawAllocMapping::new(size),
81            guard_pages,
82        };
83        stack.alloc_pages()?;
84
85        Ok(stack)
86    }
87
88    /// Create a new [`VMKernelStack`] with the default size. This function
89    /// will already allocate the backing pages for the stack.
90    ///
91    /// # Returns
92    ///
93    /// Initialized stack on success, Err(SvsmError::Mem) on error
94    pub fn new() -> Result<Self, SvsmError> {
95        VMKernelStack::new_size(STACK_SIZE)
96    }
97
98    /// Create a new [`VMKernelStack`] with the default size, packed into a
99    /// [`Mapping`]. This function / will already allocate the backing pages for
100    /// the stack.
101    ///
102    /// # Returns
103    ///
104    /// Initialized Mapping to stack on success, Err(SvsmError::Mem) on error
105    pub fn new_mapping() -> Result<Mapping, SvsmError> {
106        Ok(Mapping::new(Self::new()?))
107    }
108
109    fn alloc_pages(&mut self) -> Result<(), SvsmError> {
110        self.alloc.alloc_pages()
111    }
112}
113
114impl VirtualMapping for VMKernelStack {
115    fn mapping_size(&self) -> usize {
116        self.alloc.mapping_size() + ((self.guard_pages * 2) << PAGE_SHIFT)
117    }
118
119    fn map(&self, offset: usize) -> Option<PhysAddr> {
120        let pfn = offset >> PAGE_SHIFT;
121        let guard_offset = self.guard_pages << PAGE_SHIFT;
122
123        if pfn >= self.guard_pages {
124            self.alloc.map(offset - guard_offset)
125        } else {
126            None
127        }
128    }
129
130    fn unmap(&self, offset: usize) {
131        let pfn = offset >> PAGE_SHIFT;
132
133        if pfn >= self.guard_pages {
134            self.alloc.unmap(pfn - self.guard_pages);
135        }
136    }
137
138    fn pt_flags(&self, _offset: usize) -> PTEntryFlags {
139        PTEntryFlags::WRITABLE | PTEntryFlags::NX | PTEntryFlags::ACCESSED | PTEntryFlags::DIRTY
140    }
141}
