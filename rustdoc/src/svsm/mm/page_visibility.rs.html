26/// Makes a virtual page shared by revoking its validation, updating the
27/// page state, and modifying the page tables accordingly.
28///
29/// # Arguments
30///
31/// * `vaddr` - The virtual address of the page to be made shared.
32///
33/// # Safety
34///
35/// Converting the memory at `vaddr` must be safe within Rust's memory model.
36/// Notably any objects at `vaddr` must tolerate unsynchronized writes of any
37/// bit pattern.  In addition, the caller must take responsibility for
38/// returning a page to the private state if it is ever freed.
39unsafe fn make_page_shared(vaddr: VirtAddr) -> Result<(), SvsmError> {
40    // Revoke page validation before changing page state.
41    SVSM_PLATFORM.validate_virtual_page_range(
42        MemoryRegion::new(vaddr, PAGE_SIZE),
43        PageValidateOp::Invalidate,
44    )?;
45    let paddr = virt_to_phys(vaddr);
46    if valid_bitmap_valid_addr(paddr) {
47        valid_bitmap_clear_valid_4k(paddr);
48    }
49
50    // Ask the hypervisor to make the page shared.
51    SVSM_PLATFORM.page_state_change(
52        MemoryRegion::new(paddr, PAGE_SIZE),
53        PageSize::Regular,
54        PageStateChangeOp::Shared,
55    )?;
56
57    // Update the page tables to map the page as shared.
58    this_cpu()
59        .get_pgtable()
60        .set_shared_4k(vaddr)
61        .expect("Failed to remap shared page in page tables");
62    flush_tlb_global_sync();
63
64    Ok(())
65}

67/// Makes a virtual page private by updating the page tables, modifying the
68/// page state, and revalidating the page.
69///
70/// # Arguments
71///
72/// * `vaddr` - The virtual address of the page to be made private.
73///
74/// # Safety
75///
76/// Converting the memory at `vaddr` must be safe within Rust's memory model.
77/// No outstanding references to the page may exist.
78unsafe fn make_page_private(vaddr: VirtAddr) -> Result<(), SvsmError> {
79    // Update the page tables to map the page as private.
80    this_cpu().get_pgtable().set_encrypted_4k(vaddr)?;
81    flush_tlb_global_sync();
82
83    // Ask the hypervisor to make the page private.
84    let paddr = virt_to_phys(vaddr);
85    SVSM_PLATFORM.page_state_change(
86        MemoryRegion::new(paddr, PAGE_SIZE),
87        PageSize::Regular,
88        PageStateChangeOp::Private,
89    )?;
90
91    // Validate the page now that it is private again.
92    SVSM_PLATFORM.validate_virtual_page_range(
93        MemoryRegion::new(vaddr, PAGE_SIZE),
94        PageValidateOp::Validate,
95    )?;
96    if valid_bitmap_valid_addr(paddr) {
97        valid_bitmap_set_valid_4k(paddr);
98    }
99
100    Ok(())
101}

103/// SharedBox is a safe wrapper around memory pages shared with the host.
104pub struct SharedBox<T> {
105    ptr: NonNull<T>,
106}
107
108impl<T: FromZeros> SharedBox<T> {
109    /// Allocate some memory and share it with the host.
110    pub fn try_new_zeroed() -> Result<Self, SvsmError> {
111        let page_box = PageBox::<MaybeUninit<T>>::try_new_zeroed()?;
112        let vaddr = page_box.vaddr();
113
114        let ptr = NonNull::from(PageBox::leak(page_box)).cast::<T>();
115
116        for offset in (0..core::mem::size_of::<T>()).step_by(PAGE_SIZE) {
117            let r1 = unsafe { make_page_shared(vaddr + offset) };
118            if let Err(e1) = r1 {
119                for off in (0..offset).step_by(PAGE_SIZE) {
120                    // SAFETY: we previously marked this page as shared in this same function.
121                    let r2 = unsafe { make_page_private(vaddr + off) };
122                    if let Err(e2) = r2 {
123                        panic!(
124                            "Failed to restore page visibility ({e2:?}) after allocation failure"
125                        );
126                    }
127                }
128                // SAFETY: we previously allocated these pages in this same function
129                let _ = unsafe { PageBox::from_raw(ptr) };
130                return Err(e1);
131            }
132        }
133
134        Ok(Self { ptr })
135    }
136}
137
138impl<T> SharedBox<T> {
139    /// Returns the virtual address of the memory.
140    pub fn addr(&self) -> VirtAddr {
141        VirtAddr::from(self.ptr.as_ptr())
142    }
143
144    /// Read the currently stored value into `out`.
145    pub fn read_into(&self, out: &mut T)
146    where
147        T: FromBytes + Copy,
148    {
149        // SAFETY: `self.ptr` is valid. Any bitpattern is valid for `T`.
150        unsafe {
151            unsafe_copy_bytes(self.ptr.as_ptr(), out, 1);
152        }
153    }
154
155    /// Share `value` with the host.
156    pub fn write_from(&mut self, value: &T)
157    where
158        T: Copy + IntoBytes + Immutable,
159    {
160        // SAFETY: `self.ptr` is valid. Value can be represented as a byte slice.
161        unsafe {
162            unsafe_copy_bytes(value, self.ptr.as_ptr(), 1);
163        }
164    }
165
166    /// Leak the memory.
167    pub fn leak(self) -> NonNull<T> {
168        let ptr = self.ptr;
169        core::mem::forget(self);
170        ptr
171    }
172}

174impl<T, const N: usize> SharedBox<[T; N]> {
175    /// Clear the first `n` elements.
176    pub fn nclear(&mut self, n: usize) -> Result<(), SvsmReqError>
177    where
178        T: FromZeros,
179    {
180        if n > N {
181            return Err(SvsmReqError::invalid_parameter());
182        }
183
184        // SAFETY: `self.ptr` is valid and we did a bounds check on `n`.
185        unsafe {
186            write_bytes(self.ptr.as_ptr().cast::<T>(), n, 0);
187        }
188
189        Ok(())
190    }
191
192    /// Fill up the `outbuf` slice provided with bytes from data
193    pub fn copy_to_slice(&self, outbuf: &mut [T]) -> Result<(), SvsmReqError>
194    where
195        T: FromBytes + Copy,
196    {
197        if outbuf.len() > N {
198            return Err(SvsmReqError::invalid_parameter());
199        }
200
201        // SAFETY: `self.ptr` is valid.
202        unsafe {
203            unsafe_copy_bytes(self.ptr.as_ptr().cast(), outbuf.as_mut_ptr(), outbuf.len());
204        }
205
206        Ok(())
207    }
208}

211unsafe impl<T> Sync for SharedBox<T> where T: Sync {}

213impl<T> Drop for SharedBox<T> {
214    fn drop(&mut self) {
215        // Re-encrypt the pages.
216        let res = (0..size_of::<Self>())
217            .step_by(PAGE_SIZE)
218            .try_for_each(|offset| unsafe { make_page_private(self.addr() + offset) });
219
220        // If re-encrypting was successful free the memory otherwise leak it.
221        if res.is_ok() {
222            drop(unsafe { PageBox::from_raw(self.ptr.cast::<MaybeUninit<T>>()) });
223        } else {
224            log::error!("failed to set pages to encrypted. Memory leak!");
225        }
226    }
227}
