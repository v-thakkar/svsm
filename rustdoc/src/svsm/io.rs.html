7use core::arch::asm;
8use core::fmt::Debug;
9
10pub trait IOPort: Sync + Debug {
11    fn outb(&self, port: u16, value: u8) {
12        // SAFETY: Inline assembly to write an ioport, which does not change
13        // any state related to memory safety.
14        unsafe { asm!("outb %al, %dx", in("al") value, in("dx") port, options(att_syntax)) }
15    }
16
17    fn inb(&self, port: u16) -> u8 {
18        // SAFETY: Inline assembly to read an ioport, which does not change
19        // any state related to memory safety.
20        unsafe {
21            let ret: u8;
22            asm!("inb %dx, %al", in("dx") port, out("al") ret, options(att_syntax));
23            ret
24        }
25    }
26
27    fn outw(&self, port: u16, value: u16) {
28        // SAFETY: Inline assembly to write an ioport, which does not change
29        // any state related to memory safety.
30        unsafe { asm!("outw %ax, %dx", in("ax") value, in("dx") port, options(att_syntax)) }
31    }
32
33    fn inw(&self, port: u16) -> u16 {
34        // SAFETY: Inline assembly to read an ioport, which does not change
35        // any state related to memory safety.
36        unsafe {
37            let ret: u16;
38            asm!("inw %dx, %ax", in("dx") port, out("ax") ret, options(att_syntax));
39            ret
40        }
41    }
42
43    fn outl(&self, port: u16, value: u32) {
44        // SAFETY: Inline assembly to write an ioport, which does not change
45        // any state related to memory safety.
46        unsafe { asm!("outl %eax, %dx", in("eax") value, in("dx") port, options(att_syntax)) }
47    }
48
49    fn inl(&self, port: u16) -> u32 {
50        // SAFETY: Inline assembly to read an ioport, which does not change
51        // any state related to memory safety.
52        unsafe {
53            let ret: u32;
54            asm!("inl %dx, %eax", in("dx") port, out("eax") ret, options(att_syntax));
55            ret
56        }
57    }
58}
59
60#[derive(Default, Debug, Clone, Copy)]
61pub struct DefaultIOPort {}
62
63impl IOPort for DefaultIOPort {}
64
65pub static DEFAULT_IO_DRIVER: DefaultIOPort = DefaultIOPort {};
