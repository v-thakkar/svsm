107pub fn idt_init() {
108    // Switch #PF handler to the default one
109    idt_mut().set_entry(PF_VECTOR, IdtEntry::entry(asm_entry_pf));
110
111    // Set IST vectors
112    init_ist_vectors();
113
114    // SAFETY:
115    // Capture an address that can be used by assembly code to read the #HV
116    // doorbell page.  The address of each CPU's doorbell page may be
117    // different, but the address of the field in the PerCpu structure that
118    // holds the actual pointer is constant across all CPUs, so that is the
119    // pointer that is actually captured.  The address that is captured is
120    // stored as a usize instead of a typed value, because the declarations
121    // required for type safety here are cumbersome, and the assembly code
122    // that uses the value is not type safe in any case, so enforcing type
123    // safety on the pointer would offer no meaningful value.
124    unsafe {
125        HV_DOORBELL_ADDR = this_cpu().hv_doorbell_addr() as usize;
126    };
127}

141// Doube-Fault handler
142#[no_mangle]
143extern "C" fn ex_handler_double_fault(ctxt: &mut X86ExceptionContext) {
144    let cr2 = read_cr2();
145    let rip = ctxt.frame.rip;
146    let rsp = ctxt.frame.rsp;
147
148    if user_mode(ctxt) {
149        log::error!(
150            "Double-Fault at RIP {:#018x} RSP: {:#018x} CR2: {:#018x} - Terminating task",
151            rip,
152            rsp,
153            cr2
154        );
155        terminate();
156    } else {
157        panic!(
158            "Double-Fault at RIP {:#018x} RSP: {:#018x} CR2: {:#018x}",
159            rip, rsp, cr2
160        );
161    }
162}

199// Page-Fault handler
200#[no_mangle]
201extern "C" fn ex_handler_page_fault(ctxt: &mut X86ExceptionContext, vector: usize) {
202    let cr2 = read_cr2();
203    let rip = ctxt.frame.rip;
204    let err = ctxt.error_code;
205    let vaddr = VirtAddr::from(cr2);
206
207    if user_mode(ctxt) {
208        let kill_task: bool = if is_task_fault(vaddr) {
209            current_task()
210                .fault(vaddr, (err & PageFaultError::W.bits() as usize) != 0)
211                .is_err()
212        } else {
213            true
214        };
215
216        if kill_task {
217            log::error!("Unexpected user-mode page-fault at RIP {:#018x} CR2: {:#018x} error code: {:#018x} - Terminating task",
218                    rip, cr2, err);
219            terminate();
220        }
221    } else if this_cpu()
222        .handle_pf(
223            VirtAddr::from(cr2),
224            (err & PageFaultError::W.bits() as usize) != 0,
225        )
226        .is_err()
227        && !handle_exception_table(ctxt)
228    {
229        handle_debug_exception(ctxt, vector);
230        panic!(
231            "Unhandled Page-Fault at RIP {:#018x} CR2: {:#018x} error code: {:#018x}",
232            rip, cr2, err
233        );
234    }
235}

237// Control-Protection handler
238#[no_mangle]
239extern "C" fn ex_handler_control_protection(ctxt: &mut X86ExceptionContext, _vector: usize) {
240    // From AMD64 Architecture Programmer's Manual, Volume 2, 8.4.3
241    // Control-Protection Error Code:
242    /// A RET (near) instruction encountered a return address mismatch.
243    const NEAR_RET: usize = 1;
244    /// A RET (far) or IRET instruction encountered a return address mismatch.
245    const FAR_RET_IRET: usize = 2;
246    /// An RSTORSSP instruction encountered an invalid shadow stack restore
247    /// token.
248    const RSTORSSP: usize = 4;
249    /// A SETSSBSY instruction encountered an invalid supervisor shadow stack
250    /// token.
251    const SETSSBSY: usize = 5;
252
253    let rip = ctxt.frame.rip;
254    match ctxt.error_code & 0x7fff {
255        code @ (NEAR_RET | FAR_RET_IRET) => {
256            // Read the return address on the normal stack.
257            let ret_ptr: GuestPtr<u64> = GuestPtr::new(VirtAddr::from(ctxt.frame.rsp));
258            // SAFETY: `rsp` is a valid guest address filled by the CPU in the
259            // X86InterruptFrame
260            let ret = unsafe { ret_ptr.read() }.expect("Failed to read return address");
261
262            // Read the return address on the shadow stack.
263            let prev_rssp_ptr: GuestPtr<u64> = GuestPtr::new(VirtAddr::from(ctxt.ssp));
264            // SAFETY: `ssp` is a valid guest address filled by the CPU in the
265            // X86ExceptionContext
266            let prev_rssp = unsafe { prev_rssp_ptr.read() }
267                .expect("Failed to read address of previous shadow stack pointer");
268            // The offset to the return pointer is different for RET and IRET.
269            let offset = if code == NEAR_RET { 0 } else { 8 };
270            let ret_ptr: GuestPtr<u64> = GuestPtr::new(VirtAddr::from(prev_rssp + offset));
271            let ret_on_ssp =
272                // SAFETY: `ssp` is a valid guest address filled by the CPU in the
273                // X86ExceptionContext
274                unsafe { ret_ptr.read() }.expect("Failed to read return address on shadow stack");
275
276            panic!("thread at {rip:#018x} tried to return to {ret:#x}, but return address on shadow stack was {ret_on_ssp:#x}!");
277        }
278        RSTORSSP => {
279            panic!("rstorssp instruction encountered an unexpected shadow stack restore token at RIP {rip:#018x}");
280        }
281        SETSSBSY => {
282            panic!("setssbsy instruction encountered an unexpected supervisor shadow stack token at RIP {rip:#018x}");
283        }
284        code => unreachable!("unexpected code for #CP exception: {code}"),
285    }
286}

288// Virtualization Exception handler
289#[no_mangle]
290extern "C" fn ex_handler_ve(ctxt: &mut X86ExceptionContext) {
291    let rip = ctxt.frame.rip;
292    let code = ctxt.error_code;
293
294    if let Err(err) = handle_virtualization_exception(ctxt) {
295        log::error!("#VE handling error: {:?}", err);
296        if user_mode(ctxt) {
297            log::error!("Failed to handle #VE from user-mode at RIP {:#018x} code: {:#018x} - Terminating task", rip, code);
298            terminate();
299        } else {
300            panic!(
301                "Failed to handle #VE from kernel-mode at RIP {:#018x} code: {:#018x}",
302                rip, code
303            );
304        }
305    }
306}

328// System Call SoftIRQ handler
329#[no_mangle]
330extern "C" fn ex_handler_system_call(
331    ctxt: &mut X86ExceptionContext,
332    vector: usize,
333    event_type: IdtEventType,
334) {
335    // Ensure that this vector was not invoked as a hardware interrupt vector.
336    if event_type.is_external_interrupt(vector) {
337        panic!("Syscall handler invoked as external interrupt!");
338    }
339
340    if !user_mode(ctxt) {
341        panic!("Syscall handler called from kernel mode!");
342    }
343
344    let Ok(input) = TryInto::<u64>::try_into(ctxt.regs.rax) else {
345        ctxt.regs.rax = !0;
346        return;
347    };
348
349    ctxt.regs.rax = match input {
350        // Class 0 SysCalls.
351        SYS_EXIT => sys_exit(ctxt.regs.rdi as u32),
352        SYS_EXEC => sys_exec(ctxt.regs.rdi, ctxt.regs.rsi, ctxt.regs.r8),
353        SYS_CLOSE => sys_close(ctxt.regs.rdi as u32),
354        // Class 1 SysCalls.
355        SYS_OPEN => sys_open(ctxt.regs.rdi, ctxt.regs.rsi, ctxt.regs.r8),
356        SYS_READ => sys_read(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
357        SYS_WRITE => sys_write(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
358        SYS_SEEK => sys_seek(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
359        SYS_UNLINK => sys_unlink(ctxt.regs.rdi),
360        SYS_TRUNCATE => sys_truncate(ctxt.regs.rdi as u32, ctxt.regs.rsi),
361        SYS_OPENDIR => sys_opendir(ctxt.regs.rdi),
362        SYS_READDIR => sys_readdir(ctxt.regs.rdi as u32, ctxt.regs.rsi, ctxt.regs.r8),
363        SYS_MKDIR => sys_mkdir(ctxt.regs.rdi),
364        SYS_RMDIR => sys_rmdir(ctxt.regs.rdi),
365        // Class 3 SysCalls.
366        SYS_CAPABILITIES => sys_capabilities(ctxt.regs.rdi as u32),
367        _ => Err(SysCallError::EINVAL),
368    }
369    .map_or_else(|e| e as usize, |v| v as usize);
370}

374    let rip = ctx.frame.rip;
375    let err = ctx.error_code;

384#[no_mangle]
385pub extern "C" fn common_isr_handler_entry(vector: usize) {
386    // Since interrupt handlers execute with interrupts disabled, it is
387    // necessary to increment the per-CPU interrupt disable nesting state
388    // while the handler is running in case common code attempts to disable
389    // interrupts temporarily.  The fact that this interrupt was received
390    // means that the previous state must have had interrupts enabled.
391    let cpu = this_cpu();
392    cpu.irqs_push_nesting(true);
393
394    common_isr_handler(vector);
395
396    // Decrement the interrupt disable nesting count, but do not permit
397    // interrupts to be reenabled.  They will be reenabled during the IRET
398    // flow.
399    cpu.irqs_pop_nesting();
400}
401
402pub fn common_isr_handler(vector: usize) {
403    // Set TPR based on the vector being handled and reenable interrupts to
404    // permit delivery of higher priority interrupts.  Because this routine
405    // dispatches interrupts which should only be observable if interrupts
406    // are enabled, the IRQ nesting count must be zero at this point.
407    let previous_tpr = raw_get_tpr();
408    raw_set_tpr(tpr_from_vector(vector));
409
410    let cpu = this_cpu();
411    cpu.irqs_enable();
412
413    // Process the requested interrupt vector.
414    match vector {
415        IPI_VECTOR => this_cpu().handle_ipi_interrupt(),
416        _ => {
417            // Ignore all unrecognized interrupt vectors and treat them as
418            // spurious interrupts.
419        }
420    }
421
422    // Disable interrupts before restoring TPR.
423    cpu.irqs_disable();
424    raw_set_tpr(previous_tpr);
425
426    // Perform the EOI cycle after the interrupt processing state has been
427    // restored so that recurrent interrupts will not introduce recursion at
428    // this point.
429    apic_eoi();
430}
